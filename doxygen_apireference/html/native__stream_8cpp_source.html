60  size_t read(uint8_t* buffer, size_t bytes_requested)
61  {
62  if (!m_block)
63  {
64  if (m_stream.at_eof())
65  return 0;
66  t4::data_block block = m_stream.get_next_block();
67  if (block.get_size() == 0)
68  return 0;
69  m_block = boost::make_optional<block_data>({std::move(block), 0});
70  }
71 
72  const uint8_t* block_data = reinterpret_cast<const uint8_t*>(m_block->first.get_bytes());
73  size_t block_size = m_block->first.get_size();
74  size_t block_pos = m_block->second;
75  if (block_pos >= block_size)
76  throw std::logic_error(""); // internal error
77 
78  size_t bytes_available = block_size - block_pos;
79  if (bytes_requested >= bytes_available)
80  {
81  memcpy(buffer, block_data + block_pos, bytes_available);
82  m_block = boost::none;
83  return bytes_available;
84  }
85  else
86  {
87  memcpy(buffer, block_data + block_pos, bytes_requested);
88  m_block->second = block_pos + bytes_requested;
89  return bytes_requested;
90  }
91  }
92 };

108 jint NativeStream::read (JNIEnv* env, jbyteArray buffer, jint byteOffset, jint byteCount)
109 {
110  size_t byte_offset = t4::convert<size_t>(byteOffset);
111  size_t byte_count = t4::convert<size_t>(byteCount);
112 
113  jsize len = env->GetArrayLength(buffer);
114  jsize needed = add<jsize>(byte_offset, byte_count);
115  if (needed > len)
116  {
117  throw std::out_of_range("");
118  }
119 
120  t4::jni::java_array_elements<jbyte> byte_array(env, buffer);
121  jbyte* bytes = byte_array.writable_data();
122  if (!bytes)
123  {
124  throw std::runtime_error("");
125  }
126 
127  size_t bytes_read = m_stream->read(reinterpret_cast<uint8_t*>(bytes + byte_offset), byte_count);
128  return convert<jint>(bytes_read);
129 }
