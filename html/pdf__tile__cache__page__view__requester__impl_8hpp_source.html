46 {

49  auto req = m_requests.while_locked( [page, &pjp]( map<page_num, shared_ptr<paint_request>>& m ) -> shared_ptr<paint_request>
50  {

71 template<typename PDFTCClient>
72 void pdf_tile_cache_page_view_requester<PDFTCClient>::set_page_invisible( page_num page, update_timing timing )
73 {
74  auto req = m_requests.while_locked( [page, timing]( map<page_num, shared_ptr<paint_request>>& m ) -> std::shared_ptr<paint_request>
75  {
76  auto it = m.find( page );
77  if ( it != end( m ) )
78  {
79  switch( timing )
80  {
81  case update_timing::right_now:
82  it->second->interrupt();
83  m.erase( it );
84  return {};
85  case update_timing::after_next_incremental_update:
86  return it->second;
87  }
88  }
89 
90  return {};
91  } );
92 
93  if ( req && timing == update_timing::after_next_incremental_update )
94  req->cancel_eventually();
95 }
