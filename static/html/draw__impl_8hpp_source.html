134 template<typename Iterator>
135 void draw_annot_list( byte_transparency_group& group,
136  document& doc, dictionary const& pg,
137  affine_transform const& base_page_xform,
138  affine_transform const& page_to_device_xform,
139  int_rect const& bounds,
140  boost::optional<std::pair<Iterator, Iterator>> const& annot_range,
141  draw_parameters const& params )
142 {
143  size_t counter = 0;
144  auto draw_client = params.get<dparams::draw_client>();
145  t4::exception_ptr exception_result;
146 
147  {
148  try
149  {
150  auto& heap = *get_thread_heap();
151  transient_snapshot ss( heap );
152 
153  auto& store = doc.get_store();
154  auto& dc = doc.get_document_caches();
155  auto cs = rend::GetPageColorSpace( t3::pdf::pdfcolor::GetColorSpaceCache(dc), store, pg, group.GetColorSpace() );
156 
157  typedef typename app_traits::pdf_traits::display_traits display_traits;
158 
159  // We need to add an opacity channel if the client did not provide one and we intend to support non-normal blend modes.
160  bool needToAddOpacity = display_traits::kSupportNonNormalBlendModes && group.GetDestination().m_opacity == nullptr;
161  auto destGroup = &group;
162  bool etgNeeded = false;
163  if ( cs != group.GetColorSpace() || ( needToAddOpacity && im::GetPreferredInterleaving<app_traits>() != im::GraphicLayoutDescriptor::kChannelPlanes ) )
164  {
165  destGroup = rend::MakePageTransparencyGroup<byte_signal_traits>( group.GetConstraints(), store, pg, cs );
166  etgNeeded = true;
167  }
168  else if ( needToAddOpacity )
169  {
170  // If we get here the client provided a TransparencyGroup without an opacity channel but we need one in case
171  // we hit a blend mode, AND the preferred interleaving is kChannelPlanes. Under these conditions we can simply
172  // create a new TranpsarencyGroup drawing directly into group's color buffer, but with an opacity buffer allocated
173  // on the side.
174  tetraphilia_assert( cs == group.GetColorSpace() );
175  tetraphilia_assert( im::GetPreferredInterleaving<app_traits>() == im::GraphicLayoutDescriptor::kChannelPlanes );
176  im::PixelBufferLayoutDescriptor qLayout = im::MakePreferredAlphaLayoutDesc( false, false );
177  im::TransientPixelStore<byte_signal_traits> store2( group.GetConstraints().GetAlignedInterestBounds(), qLayout, true, heap );
178  im::GraphicLayoutDescriptor desc = im::MakeGraphicLayoutDescriptor( cs->GetNumSignals(),
179  im::GraphicLayoutDescriptor::kShapeIgnored,
180  im::GraphicLayoutDescriptor::kOpacityReplicated,
181  im::GraphicLayoutDescriptor::kColorPerChannel );
182 
183  typedef im::PixelBuffer<byte_signal_traits> pixel_buffer_type;
184  typedef typename im::RasterDestinationType<byte_signal_traits>::type raster_destination_type;
185  raster_destination_type dst = group.GetDestination();
186  pixel_buffer_type* q = t4::make_transient<pixel_buffer_type>( heap, store2.GetPixelBuffer() );
187  dst.m_opacity = q;
188  destGroup = t4::make_transient<byte_transparency_group>( heap, group.GetConstraints(), dst, cs, desc, group.GetBackdrop(), true, false );
189  }
190 
191  draw_annotations_list( doc, pg, counter, base_page_xform, page_to_device_xform, *destGroup, annot_range, params );
192  if ( etgNeeded )
193  {
194  typedef im::HardClip<byte_signal_traits> hard_clip_type;
195  hard_clip_type hc( group.GetConstraints().m_interestBounds );
196  auto identity = im::MakeIdentityMasker<byte_signal_traits>( group.GetConstraints() );
197  group.EndTransparencyGroup( group.GetConstraints(), &hc, identity, identity, destGroup );
198  }
199  }
200  catch( thread_interrupted )
201  {
202  throw;
203  }
204  catch( ... )
205  {
206  exception_result = current_exception();
207  }
208  }
209 
210  draw_client.drawing_finished( ++counter, group, exception_result );
211 }
