41 template <typename StreamCopyStrategy>
42 reference copy( default_ref_copy_strategy& ths, const reference& ref, const schema& schema, StreamCopyStrategy&& stream_copy_strategy )
43 {
44  if ( ths.known_references.count( ref ) > 0 )
45  {
46  return ths.known_references[ref];
47  }
48 
49  namespace t3_store = t3::pdf::store;
50 
51  auto& dest_store = ths.dest_store;
52  auto obj = ths.src_store.ResolveReference( ref );
53 
54  switch ( obj.Type() )
55  {
56  case t3_store::kObjectTypeNull:
57  return dest_store.CreateNewIndirectNull();
58 
59  case t3_store::kObjectTypeBool:
60  return dest_store.CreateNewIndirectBoolean( obj.BoolValue_NoTypeCheck() );
61 
62  case t3_store::kObjectTypeInteger:
63  return dest_store.CreateNewIndirectInteger( obj.IntegerValue_NoTypeCheck() );
64 
65  case t3_store::kObjectTypeReal:
66  return dest_store.CreateNewIndirectReal( obj.RealValue_NoTypeCheck() );
67 
68  case t3_store::kObjectTypeName:
69  return dest_store.CreateNewIndirectName( { obj.NameValue_NoTypeCheck().get() } );
70 
71  case t3_store::kObjectTypeString:
72  {
73  auto src_str = obj.StringValue_NoTypeCheck();
74  size_t len = static_cast<size_t>( src_str.end() - src_str.begin() );
75 
76  return dest_store.CreateNewIndirectString( { src_str.begin(), len } );
77  }
78 
79  case t3_store::kObjectTypeArray:
80  {
81  auto dest_array_ref = dest_store.CreateNewIndirectArray();
82  ths.known_references.emplace( ref, dest_array_ref );
83 
84  auto dest_array = dest_store.ResolveReference( dest_array_ref ).ArrayValue_NoTypeCheck();
85  copy_array( obj.ArrayValue_NoTypeCheck(), dest_array, schema, ths, stream_copy_strategy );
86 
87  return dest_array_ref;
88  }
89 
90  case t3_store::kObjectTypeDictionary:
91  {
92  auto dest_dict_ref = dest_store.CreateNewIndirectDictionary();
93  ths.known_references.emplace( ref, dest_dict_ref );
94 
95  auto dest_dict = dest_store.ResolveReference( dest_dict_ref ).DictionaryValue_NoTypeCheck();
96  copy_dictionary( obj.DictionaryValue_NoTypeCheck(), dest_dict, schema, ths, stream_copy_strategy );
97 
98  return dest_dict_ref;
99  }
100 
101  case t3_store::kObjectTypeLargeInteger:
102  return dest_store.CreateNewIndirectLargeInteger( obj.LargeIntegerValue_NoTypeCheck() );
103 
104  case t3_store::kObjectTypeReference:
105  throw t4::corrupt_data();
106  }
107 }

110 template <typename PDFContainer, typename Key, typename ReferenceCopyStrategy, typename StreamCopyStrategy>
111 void put_object_copy_in_container( const Key key, const object& obj, PDFContainer& pdf_container, const schema& schema,
112  ReferenceCopyStrategy&& ref_copy_strategy, StreamCopyStrategy&& stream_copy_strategy)
113 {
114  namespace t3_store = t3::pdf::store;
115 
116  switch ( obj.Type() )
117  {
118  case t3_store::kObjectTypeNull:
119  pdf_detail::put_null_in_container( pdf_container, key );
120  break;
121 
122  case t3_store::kObjectTypeBool:
123  pdf_container.PutBoolean( key, obj.BoolValue_NoTypeCheck() );
124  break;
125 
126  case t3_store::kObjectTypeInteger:
127  pdf_container.PutInteger( key, obj.IntegerValue_NoTypeCheck() );
128  break;
129 
130  case t3_store::kObjectTypeReal:
131  pdf_container.PutReal( key, obj.RealValue_NoTypeCheck() );
132  break;
133 
134  case t3_store::kObjectTypeName:
135  pdf_container.PutName( key, obj.NameValue_NoTypeCheck().get() );
136  break;
137 
138  case t3_store::kObjectTypeString:
139  {
140  auto src_string = obj.StringValue_NoTypeCheck();
141  size_t len = static_cast<size_t>( src_string.end() - src_string.begin() );
142  pdf_container.PutString( key, { src_string.begin(), len } );
143  break;
144  }
145 
146  case t3_store::kObjectTypeLargeInteger:
147  pdf_container.PutLargeInteger( key, obj.LargeIntegerValue_NoTypeCheck() );
148  break;
149 
150  case t3_store::kObjectTypeArray:
151  {
152  auto new_array = pdf_detail::create_new_array_in_container( pdf_container, key );
153  copy_array( obj.ArrayValue_NoTypeCheck(), new_array, schema, ref_copy_strategy, stream_copy_strategy );
154  break;
155  }
156 
157  case t3_store::kObjectTypeDictionary:
158  {
159  auto new_dict = pdf_detail::create_new_dictionary_in_container( pdf_container, key );
160  copy_dictionary( obj.DictionaryValue_NoTypeCheck(), new_dict, schema, ref_copy_strategy, stream_copy_strategy );
161  break;
162  }
163 
164  case t3_store::kObjectTypeReference:
165  {
166  reference ref = copy( ref_copy_strategy, obj.ReferenceValue_NoTypeCheck(), schema, stream_copy_strategy );
167  pdf_container.PutReference( key, ref );
168  break;
169  }
170  }
171 }

174 template <typename ReferenceCopyStrategy, typename StreamCopyStrategy>
175 void copy_array( const array& src, array& dest, const schema& schema, ReferenceCopyStrategy&& ref_copy_strategy, StreamCopyStrategy&& stream_copy_strategy )
176 {
177  size_t i = 0;
178  for ( auto iter = src.begin_noresolve(); iter != src.end_noresolve(); ++iter, ++i )
179  {
180  put_object_copy_in_container( i, *iter, dest, schema.get_schema_for_key( i ), ref_copy_strategy, stream_copy_strategy );
181  }
182 }

185 template <typename ReferenceCopyStrategy, typename StreamCopyStrategy>
186 void copy_dictionary( const dictionary& src, dictionary& dest, const schema& schema, ReferenceCopyStrategy&& ref_copy_strategy, StreamCopyStrategy&& stream_copy_strategy )
187 {
188  for ( auto iter = src.begin_noresolve(); iter != src.end_noresolve(); ++iter )
189  {
190  auto key = iter->GetKey();
191  if ( schema.skip_key( key ) )
192  {
193  continue;
194  }
195 
196  put_object_copy_in_container( key, iter->GetValue(), dest, schema.get_schema_for_key( key ), ref_copy_strategy, stream_copy_strategy );
197  }
198 
199  if ( HasStream( src ) )
200  {
201  copy_stream( stream_copy_strategy, src, dest );
202  }
203 }
