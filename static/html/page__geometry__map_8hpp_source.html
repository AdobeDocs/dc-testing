82  {}
83 
84  void set_page_geometry( pdf::page_num page_num, page_geometry const& geom )
85  {
86  auto known_pred = []( bool known ){ return known; };
87  pdf::page_num next_known = m_knowns.find( page_num + 1, std::numeric_limits<pdf::page_num>::max(), known_pred );
88  m_geometry.set_range( page_num, next_known, geom );
89  pdf::page_num prev_known = m_knowns.find( 0, page_num, known_pred );
90  if ( prev_known == page_num )
91  m_geometry.set_range( 0, page_num, geom );
92  m_knowns.set( page_num, true );
93  }
94 
95  std::pair<bool, page_geometry const&> get_page_geometry( pdf::page_num page_num ) const
96  {
97  return std::make_pair( m_knowns[ page_num ], std::cref( m_geometry[ page_num ] ) );
98  }
99 
100  void update_page_geometry( pdf::page_num page_num );
101 };
