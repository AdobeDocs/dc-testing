32 public class AESCryptor

34  //The block chaining part (of CBC mode algorithm) has been taken care as mentioned in the comment below, Hence suppressing this warning
35  @SuppressLint("GetInstance")
36  public boolean Init_Decryptor(byte[] keyData){
37  try
38  {
39  SecretKey key = new SecretKeySpec(keyData, CIPHER_ALGORITHM);
40  mDecryptor = Cipher.getInstance(DECRYPTOR_TRANSFORMATION);
41  mDecryptor.init(Cipher.DECRYPT_MODE, key);
42  }
43  catch (Exception e)
44  {
45  // "ARAESCryptor: In Init_Decryptor - EXCEPTION: "
46  return false;
47  }
48  int block_size = mDecryptor.getBlockSize();
49  if(AES_BLOCK_SIZE != block_size)
50  {
51  // "ARAESCryptor: In Init_Decryptor - Incorrect block size: "
52  return false;
53  }
54  return true;
55  }

57  public boolean Init_Encryptor(byte[] keyData, byte[] iv, boolean PKCS5Padding){
58  try
59  {
60  SecretKey key = new SecretKeySpec(keyData, CIPHER_ALGORITHM);
61  if (PKCS5Padding)
62  mEncryptor = Cipher.getInstance(ENCRYPTOR_TRANSFORMATION_PADDING);
63  else
64  mEncryptor = Cipher.getInstance(ENCRYPTOR_TRANSFORMATION);
65  AlgorithmParameterSpec paramSpec = new IvParameterSpec(iv);
66  mEncryptor.init(Cipher.ENCRYPT_MODE, key, paramSpec);
67  }
68  catch (Exception e)
69  {
70  // "ARAESCryptor: In Init_Encryptor - EXCEPTION: "
71  return false;
72  }
73  int block_size = mEncryptor.getBlockSize();
74  if(AES_BLOCK_SIZE != block_size)
75  {
76  // "ARAESCryptor: In Init_Encryptor - Incorrect block size: "
77  return false;
78  }
79  return true;
80  }

82  public void Decrypt(byte[] input, int inputLen, byte[] output){
83  int outSize = 0;
84  try
85  {
86  outSize = mDecryptor.update(input, 0, inputLen, output);
87  }
88  catch (Exception e)
89  {
90  // "AESCrypt: In Decrypt - EXCEPTION: "
91  output = null;
92  }
93 
94  if(AES_BLOCK_SIZE != outSize)
95  {
96  // "AESCrypt: In Decrypt - Incorrect output size: "
97  output = null;
98  }
99  }

101  public int Encrypt(byte[] input, int inputLen, byte[] output, int outBufSize)
102  {
103  int outputLen = mEncryptor.getOutputSize(inputLen);
104  if(outBufSize < outputLen)
105  {
106  // AESCrypt: In Encrypt - outBufSize is less than outputLen"
107  return -1;
108  }
109  try
110  {
111  outputLen = mEncryptor.doFinal(input, 0, inputLen, output);
112  }
113  catch (Exception e)
114  {
115  // "AESCrypt: In Encrypt - EXCEPTION: "
116  output = null;
117  return -1;
118  }
119 
120  return outputLen;
121  }

126  /* The block chaining part (of CBC mode algorithm) has been taken care of in the calling
