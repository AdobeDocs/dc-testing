316 static real_rect create_from_glyph(const extracted_glyph & g)
317 {
318  auto leading = g.GetLeadingEdge();
319  auto trailing = g.GetTrailingEdge();
320  auto xRange = minmax( get_x(leading.first), get_x(leading.second), get_x(trailing.first), get_x(trailing.second) );
321  auto yRange = minmax( get_y(leading.first), get_y(leading.second), get_y(trailing.first), get_y(trailing.second) );
322  return make<real_rect>( xRange.first, yRange.first, xRange.second, yRange.second );
323 }

325 static t4::vector<t4::pair<t4::string, t4::real_rect>> Document_nativeGetPageTextGeometry( JNIEnv* env, t4::jni::native_ref<doc_wrapper> doc, t4::pdf::page_num pageNum )
326 {
327  shared_ref<document> ptr_doc = get_doc_ref(doc);
328  context_lock lock;
329 
330  auto & doc_ref = *ptr_doc;
331  auto page_range = whole_page_content_range(doc_ref, pageNum);
332 
333  t4::vector<boost::uint32_t> word_vec;
334  t4::real_rect word_rect;
335 
336  t4::vector<t4::pair<t4::string, t4::real_rect>> word_coordinates;
337 
338  auto f = [&]( std::uint32_t u, extracted_glyph g )
339  {
340  bool whitespace = isWhitespace(u);
341  bool shouldBreak = whitespace; // || g.StartsNewRange()?, breaks too often
342  if (shouldBreak && !word_vec.empty())
343  {
344  word_coordinates.push_back(std::make_pair(t4::pdf::to_utf8(word_vec), word_rect));
345  word_vec.clear();
346  }
347 
348  if (!whitespace)
349  {
350  t4::real_rect glyph_rect = create_from_glyph(g);
351  if (word_vec.empty())
352  {
353  word_rect = glyph_rect;
354  } else {
355  word_rect = t4::union_(word_rect, glyph_rect);
356  }
357  word_vec.push_back(u);
358  }
359  };
360 
361  t4::pdf::enumerate_content(doc_ref, page_range, std::move(f));
362 
363  // add left over word if not empty
364  if (!word_vec.empty())
365  {
366  word_coordinates.push_back(std::make_pair(t4::pdf::to_utf8(word_vec), word_rect));
367  }
368 
369  return word_coordinates;
370 }

413 static intersection_details getIntersectionNode(t4::shared_ptr<t4::pdf::document> ptr_doc, const t4::vector<android_accessible>& accessibleContentNodes, android_content_point android_point, t4::pdf::content_order order)
414 {
415  intersection_details intersectionDetails;
416 
417  //conversion from <t4::pdf::android_content_point> to <t4::pdf::content_point>
418  content_point point(t4::pdf::get_page_id( *ptr_doc, android_point.m_page_index),
419  android_point.m_dlentry_num,
420  android_point.m_string_num,
421  android_point.m_glyph_num,

423  android_point.m_total_num_unicode,

426  auto size = accessibleContentNodes.size();
427 
428  for (size_t i = 0; i < size; i++) {
429  if (accessibleContentNodes.at(i).m_content_range.Contains(point)) {
430  /*
431  * if we find intersection node, we have to find offset of start point from beginning of current node
432  */
433 
434  intersectionDetails.m_block_index = static_cast<int>(i); //the index of accessibility node in which selected point lies
435  //creating contentRange from beginning of node to selected point
436  t4::pdf::content_range contentRange = accessibleContentNodes.at(i).m_content_range;
437  contentRange.m_to = point;
438 
439  //iterating over all content points within contentRange to calculate total offset
440  auto charCount = 0;
441  t4::real_rect glyph_rect;
442  auto count_chars = [&charCount,&glyph_rect](std::uint32_t u, extracted_glyph g) {
443  content_point current_point = g.ContentPoint();
444  if(current_point.m_totalNumUnicode>0) charCount ++;
445  glyph_rect = create_from_glyph(g);
446  };
447  t4::pdf::enumerate_content(*ptr_doc, contentRange, std::move(count_chars), order);
448  intersectionDetails.m_len = charCount;
449  intersectionDetails.m_rect = glyph_rect;
450  break;
451  }
452  }
453 
454  return intersectionDetails;
455 }

457 void createHierarchyOfAccessibleNodes(aaa::accessible_content_result content,
458  android_accessible_result& res)
459 {
460  t4::vector<t4::aaa::accessible> accessibleContentNodes = content.nodes;
461 
462  t4::vector<android_accessible> result_accessibleNodes;
463  t4::vector<t4::aaa::accessible_list> accessible_list;
464  std::vector<boost::shared_ptr<t4::aaa::accessible>> accessibleNodes;
465 
466  for (auto it = accessibleContentNodes.begin(); it != accessibleContentNodes.end(); it++)
467  {
468  accessibleNodes.push_back(boost::make_shared<t4::aaa::accessible>(*it));
469  }
470  auto accessibleNodes_size = accessibleNodes.size();
471 
472  for (size_t i = 0; i < accessibleNodes.size(); i++)
473  {
474  auto accessibleNode = accessibleNodes.at(i);
475  android_accessible accessible;
476 
477  size_t row_count = 0;
478  size_t column_count = 0;
479  size_t row_index = 0;
480  size_t column_index = 0;
481  size_t parent_id = 0;
482  size_t contentRoleSecondary = 0;
483 
484  auto contentRolePrimary = accessibleNode->m_role;
485 
486  try
487  {
488  if(content.make_accessible)
489  {
490  t4::aaa::make_accessible make_accessible = *(content.make_accessible);
491  switch(contentRolePrimary)
492  {
493  case t4::aaa::role::statictext:
494  {
495  auto header_node = make_accessible.get_heading_level(*accessibleNode);
496  auto link_node = make_accessible.get_link_ancestor(*accessibleNode);
497  auto list_node = make_accessible.get_list_ancestor(*accessibleNode);
498 
499  if (header_node)
500  {
501  contentRoleSecondary = contentRoleSecondary | static_cast<int>(t4::aaa::role::heading);
502  }
503 
504  if (link_node)
505  {
506  contentRoleSecondary = contentRoleSecondary | static_cast<int>(t4::aaa::role::link);
507  }
508 
509  if (list_node)
510  {
511  contentRoleSecondary = contentRoleSecondary | static_cast<int>(t4::aaa::role::listitem);
512 
513  // computing child node of this list_item_node
514  auto list_item_node = make_accessible.get_list_item_ancestor(*accessibleNode);
515 
516  if (list_item_node)
517  row_index = list_item_node->get_index_in_list();
518 
519  row_count = list_node->get_item_count();
520 
521  auto iterator = std::find(accessible_list.begin(), accessible_list.end(), *list_node);
522  if (iterator == accessible_list.end())
523  {
524  accessibleNodes.push_back(boost::make_shared<t4::aaa::accessible_list>(*list_node));
525 
526  accessible_list.push_back(*list_node);
527 
528  parent_id = accessible_list.size() + accessibleNodes_size;
529  }
530  else
531  {
532  parent_id = static_cast<size_t>( std::distance(accessible_list.begin(), iterator )) + accessibleNodes_size + 1;
533  }
534  }
535  break;
536  }
537 
538  case t4::aaa::role::list:
539  {
540  contentRoleSecondary = contentRoleSecondary | static_cast<int>(t4::aaa::role::list);
541  auto accessible_list_node = static_cast<t4::aaa::accessible_list&>(*accessibleNode);
542  row_count = accessible_list_node.get_item_count();
543  column_count = 0;
544  break;
545  }
546 
547  default:
548  {
549  // extend this list to define further roles.
550  }
551  }
552  }
553  }
554  catch (const std::exception&) { /*no op*/ }
555 
556  auto contentDesc = accessibleNode->m_screen_text;
557  if (contentDesc.empty())
558  {
559  contentDesc = accessibleNode->m_spoken_text;
560  }
561 
562  accessible.m_row_count = row_count;
563  accessible.m_column_count = column_count;
564  accessible.m_row_index = row_index;
565  accessible.m_column_index = column_index;
566  accessible.m_parent_id = parent_id;
567  accessible.m_content_role_secondary = contentRoleSecondary;
568  accessible.id = i + 1;
569  accessible.m_content_role_primary = static_cast<size_t>(accessibleNode->m_role);
570  accessible.m_content_box = accessibleNode->m_rect;
571  accessible.m_content_desc = contentDesc;
572  accessible.m_content_value = accessibleNode->m_value;
573  accessible.m_content_range = accessibleNode->m_range;
574  result_accessibleNodes.push_back(accessible);
575  }
576 
577  res.m_accessible_nodes = result_accessibleNodes;
578 }
